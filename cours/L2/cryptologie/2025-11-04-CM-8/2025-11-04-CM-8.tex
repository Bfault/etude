\begin{proposition}
    La connaissance de $\phi$ et $n$ permet de retrouver $p$ et $q$.
\end{proposition}

\begin{demonstration}
    \begin{align*}
        n - \phi + 1 &= pq - (p-1)(q-1) + 1 \\
        &= pq - (pq - p - q + 1) + 1 \\
        &= p + q
    \end{align*}

    On a aussi $n = pq$.

    $$
    (X - p)(X - q) = X^2 - (p + q)X + pq

\end{demonstration}
